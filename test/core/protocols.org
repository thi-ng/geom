    #{:math :det :inv :rotate :scale :tx :vtx})

  #?(:clj
     (defn satisfies-all?
       [type & protos]
       (testing
           (.getName type)
         (doseq [p protos]
           (is (true? (extends? (proto-ids p) type))
               (str (get-in proto-ids [p :on-interface])))))))
